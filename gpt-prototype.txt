      if(cls) x.className = cls;
      if(html !== undefined) x.innerHTML = html;

      const card = el('article','card work-card');
      const badge = el('div','badge', item._kind);
      const thumb = el('div','thumb');
      const img = new Image(); img.alt = item.title; img.src = item.image || 'assets/placeholder.png';

    const toggleModal = (on) => {
      modal.classList.toggle('open', !!on);
      modal.setAttribute('aria-hidden', on ? 'false' : 'true');
      if(on) document.body.style.overflow='hidden'; else document.body.style.overflow='';
    };
    openBtn.addEventListener('click', (e)=>{ e.stopPropagation(); toggleModal(true); });
    closeBtn.addEventListener('click', ()=> toggleModal(false));
    modal.addEventListener('click', (e)=>{ if(e.target === modal) toggleModal(false); });
